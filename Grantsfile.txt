        <div class="card">
            <h2>Create Team</h2>

            <div class="primary-section">
                <h3>Primary Spirit</h3>
                <input type="text" id="primaryName" placeholder="Primary Spirit Name">
                <div class="slot-controls">
                    <span>Slots:</span>
                    <button class="slot-btn active" onclick="setPrimarySlots(1)">1</button>
                    <button class="slot-btn" onclick="setPrimarySlots(2)">2</button>
                    <button class="slot-btn" onclick="setPrimarySlots(3)">3</button>
                </div>
            </div>

            <div class="support-section">
                <div class="header-row">
                    <h3>Support Spirits</h3>
                    <div class="slot-indicator" id="slotIndicator"></div>
                </div>
                <div id="supportsList"></div>
                <button class="btn btn-primary" onclick="addSupport()">+ Add Support Spirit</button>
            </div>

            <button class="btn btn-success" onclick="saveTeam()">üíæ Save Team</button>
        </div>

        <div class="card" id="savedTeamsCard">
            <h2>Saved Teams</h2>
            <div id="savedTeamsList"></div>
        </div>
    </div>

    <script>
        let currentTeam = {
            name: '',
            primary: { name: '', slots: 1 },
            supports: []
        };

        let teams = [];

        // Load teams from localStorage
        function loadTeams() {
            const saved = localStorage.getItem('spirit-teams');
            if (saved) {
                teams = JSON.parse(saved);
                renderSavedTeams();
            }
        }

        // Save teams to localStorage
        function saveTeamsToStorage() {
            localStorage.setItem('spirit-teams', JSON.stringify(teams));
        }

        function setPrimarySlots(slots) {
            currentTeam.primary.slots = slots;
            currentTeam.supports = [];
            
            document.querySelectorAll('.primary-section .slot-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
            
            renderSupports();
            updateSlotIndicator();
        }

        function addSupport() {
            if (currentTeam.supports.length < 3 && getTotalSlotsUsed() < currentTeam.primary.slots) {
                currentTeam.supports.push({ name: '', slots: 1 });
                renderSupports();
                updateSlotIndicator();
            }
        }

        function removeSupport(index) {
            currentTeam.supports.splice(index, 1);
            renderSupports();
            updateSlotIndicator();
        }

        function setSupportSlots(index, slots) {
            currentTeam.supports[index].slots = slots;
            renderSupports();
            updateSlotIndicator();
        }

        function getTotalSlotsUsed() {
            return currentTeam.supports.reduce((sum, s) => sum + s.slots, 0);
        }

        function updateSlotIndicator() {
            const indicator = document.getElementById('slotIndicator');
            const total = currentTeam.primary.slots;
            const used = getTotalSlotsUsed();
            
            indicator.innerHTML = '';
            for (let i = 0; i < total; i++) {
                const box = document.createElement('div');
                box.className = 'slot-box' + (i < used ? ' filled' : '');
                indicator.appendChild(box);
            }
        }

        function renderSupports() {
            const list = document.getElementById('supportsList');
            list.innerHTML = '';
            
            currentTeam.supports.forEach((support, index) => {
                const div = document.createElement('div');
                div.className = 'support-item';
                
                const slotsUsed = getTotalSlotsUsed();
                const currentSlots = support.slots;
                
                div.innerHTML = `
                    <input type="text" placeholder="Support Spirit Name" value="${support.name}" 
                           onchange="currentTeam.supports[${index}].name = this.value">
                    <div class="support-controls">
                        <div class="slot-controls">
                            <span>Slots:</span>
                            <button class="slot-btn ${support.slots === 1 ? 'active blue' : ''}" 
                                    onclick="setSupportSlots(${index}, 1)">1</button>
                            <button class="slot-btn ${support.slots === 2 ? 'active blue' : ''}" 
                                    onclick="setSupportSlots(${index}, 2)"
                                    ${slotsUsed - currentSlots + 2 > currentTeam.primary.slots ? 'disabled' : ''}>2</button>
                            <button class="slot-btn ${support.slots === 3 ? 'active blue' : ''}" 
                                    onclick="setSupportSlots(${index}, 3)"
                                    ${slotsUsed - currentSlots + 3 > currentTeam.primary.slots ? 'disabled' : ''}>3</button>
                        </div>
                        <button class="btn-danger" onclick="removeSupport(${index})">üóëÔ∏è</button>
                    </div>
                `;
                list.appendChild(div);
            });
        }

        function saveTeam() {
            const teamName = document.getElementById('teamName').value;
            const primaryName = document.getElementById('primaryName').value;
            
            if (!teamName || !primaryName) {
                alert('Please enter team name and primary spirit name');
                return;
            }
            
            teams.push({
                id: Date.now(),
                name: teamName,
                primary: { 
                    name: primaryName, 
                    slots: currentTeam.primary.slots 
                },
                supports: [...currentTeam.supports]
            });
            
            saveTeamsToStorage();
            renderSavedTeams();
            
            // Reset form
            document.getElementById('teamName').value = '';
            document.getElementById('primaryName').value = '';
            currentTeam = {
                name: '',
                primary: { name: '', slots: 1 },
                supports: []
            };
            renderSupports();
            updateSlotIndicator();
        }

        function deleteTeam(id) {
            if (confirm('Delete this team?')) {
                teams = teams.filter(t => t.id !== id);
                saveTeamsToStorage();
                renderSavedTeams();
            }
        }

        function renderSavedTeams() {
            const list = document.getElementById('savedTeamsList');
            const card = document.getElementById('savedTeamsCard');
            
            if (teams.length === 0) {
                card.classList.add('hidden');
                return;
            }
            
            card.classList.remove('hidden');
            list.innerHTML = '';
            
            teams.forEach(team => {
                const div = document.createElement('div');
                div.className = 'saved-team';
                
                let supportsHTML = '';
                if (team.supports.length > 0) {
                    const supportRows = team.supports.map(s => {
                        const slots = Array(s.slots).fill(0).map(() => '<div class="slot-box filled"></div>').join('');
                        return `
                            <div class="support-row">
                                <p style="color: #1f2937;">${s.name}</p>
                                <div class="slot-indicator">${slots}</div>
                            </div>
                        `;
                    }).join('');
                    
                    supportsHTML = `
                        <div class="team-supports">
                            <p>Supports</p>
                            ${supportRows}
                        </div>
                    `;
                }
                
                const primarySlots = Array(team.primary.slots).fill(0).map(() => '<div class="slot-box filled"></div>').join('');
                
                div.innerHTML = `
                    <div class="saved-team-header">
                        <h3>${team.name}</h3>
                        <button class="btn-danger" onclick="deleteTeam(${team.id})">üóëÔ∏è</button>
                    </div>
                    <div class="team-primary">
                        <p>Primary</p>
                        <p style="color: #1f2937; margin-bottom: 4px;">${team.primary.name}</p>
                        <div class="slot-indicator">${primarySlots}</div>
                    </div>
                    ${supportsHTML}
                `;
                list.appendChild(div);
            });
        }

        // Initialize
        loadTeams();
        updateSlotIndicator();
    </script>
